# function to calculate value weighted return
def wavg(group, avg_name, weight_name):
    d = group[avg_name]
    w = group[weight_name]
    try:
        return (d * w).sum() / w.sum()
    except ZeroDivisionError:
        return np.nan

## create function to get regression results

def PortReg(df_reg, dt_start, dt_end): 
    # get subsample from dt_start to dt_end
    df_reg_sub = df_reg[(df_reg['date']>=dt_start) & (df_reg['date']<=dt_end)].copy()

    # create a df to store regression results
    sbport_list = df_reg['sbport'].unique()
    df_res = pd.DataFrame({'sbport': sbport_list, 
                           'b': np.nan, 't_b': np.nan, 
                           's': np.nan, 't_s': np.nan, 
                           'h': np.nan, 't_h': np.nan, 
                           'r2': np.nan, 'se': np.nan})

    # run regression
    for sbport in list(sbport_list):
        # get get X and y in regression
        est_sub = (df_reg_sub['sbport']==sbport)
        X = df_reg_sub[['MyMKTRF', 'MySMB', 'MyHML']][est_sub]
        y = df_reg_sub['ret_rf'][est_sub]

        # fit an OLS model with intercept on MKTRF, SMB, HML
        X = sm.add_constant(X)
        est = sm.OLS(y, X).fit()

        # get estimated parameters
        df_res.loc[df_res['sbport']==sbport, 'b'] = est.params.__getitem__('MyMKTRF')
        df_res.loc[df_res['sbport']==sbport, 't_b'] = est.tvalues['MyMKTRF']
        df_res.loc[df_res['sbport']==sbport, 's'] = est.params.__getitem__('MySMB')
        df_res.loc[df_res['sbport']==sbport, 't_s'] = est.tvalues['MySMB']
        df_res.loc[df_res['sbport']==sbport, 'h'] = est.params.__getitem__('MyHML')
        df_res.loc[df_res['sbport']==sbport, 't_h'] = est.tvalues['MyHML']
        df_res.loc[df_res['sbport']==sbport, 'r2'] = est.rsquared
        df_res.loc[df_res['sbport']==sbport, 'se'] = np.sqrt(est.mse_resid)
        
    # return the estimation results
    return df_res
